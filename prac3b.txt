using System;
using System.Web.UI;
namespace Practical1
{
public partial class Operations2 : System.Web.UI.Page
{
protected void BtnOperation2_Click(object sender, EventArgs e)
{
int choice = int.Parse(ddlOption.SelectedValue);
string input = userinput.Text;
lblResult.Text = "";

        switch (choice)
        {
            case 1:
                int num = int.Parse(input);
                long fact = 1;

                for (int i = 1; i <= num; i++)
                {
                    fact *= i;
                }

                lblResult.Text = "Factorial = " + fact;
                break;

            case 2:
                double inr = double.Parse(input);
                double dollarRate = 89.36;
                double usd = inr / dollarRate;

                lblResult.Text = "USD = " + usd.ToString("0.00");
                break;

            case 3:
                string[] coeff = input.Split(',');

                if (coeff.Length != 3)
                {
                    lblResult.Text = "Enter input as: a,b,c";
                    break;
                }
                double a = double.Parse(coeff[0]);
                double b = double.Parse(coeff[1]);
                double c = double.Parse(coeff[2]);
                double d = (b * b) - (4 * a * c);

                if (d > 0)
                {
                    double r1 = (-b + Math.Sqrt(d)) / (2 * a);
                    double r2 = (-b - Math.Sqrt(d)) / (2 * a);

                    lblResult.Text = $"Real Roots: {r1} and {r2}";
                }
                else if (d == 0)
                {
                    double r = -b / (2 * a);
                    lblResult.Text = "One Real Root: " + r;
                }
                else
                {
                    lblResult.Text = "Imaginary Roots";
                }
                break;
            case 4:
                double celsius = double.Parse(input);
                double fahrenheit = (celsius * 9 / 5) + 32;
                lblResult.Text = "Fahrenheit = " + fahrenheit;
                break;
            default:
                lblResult.Text = "Invalid Option";
                break;
        }
    }
}
}
