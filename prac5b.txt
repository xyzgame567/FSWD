using System;
using System.Data;
using System.Data.SqlClient;
using System.Web.UI;
using System.Web.UI.WebControls;
namespace Practical1
{
public partial class DisconnectedDBcrud : Page
{
SqlConnection con = new SqlConnection(
System.Configuration.ConfigurationManager.ConnectionStrings["MyConn"].ConnectionString);
SqlDataAdapter da;
DataSet ds = new DataSet();
protected void Page_Load(object sender, EventArgs e)
{
if (!IsPostBack)
{
BindGrid();
}
}
void BindGrid()
{
da = new SqlDataAdapter("SELECT * FROM RegisterTable", con);
ds.Clear();
da.Fill(ds, "RegisterTable");
// Set primary key for Find method
ds.Tables["RegisterTable"].PrimaryKey = new DataColumn[] { ds.Tables["RegisterTable"].Columns["Id"] };
GridView1.DataSource = ds.Tables["RegisterTable"];
GridView1.DataBind();
}
protected void btnInsert_Click(object sender, EventArgs e)
{
da = new SqlDataAdapter("SELECT * FROM RegisterTable", con);
SqlCommandBuilder cb = new SqlCommandBuilder(da);
ds.Clear();
da.Fill(ds, "RegisterTable");
DataRow dr = ds.Tables["RegisterTable"].NewRow();
dr["Name"] = txtname.Text;
dr["Age"] = txtage.Text;
dr["Phone"] = txtnumber.Text;
dr["Email"] = txtemail.Text;
ds.Tables["RegisterTable"].Rows.Add(dr);
da.Update(ds, "RegisterTable");
BindGrid();
ClearFields();
}
protected void btnUpdate_Click(object sender, EventArgs e)
{
if (GridView1.SelectedIndex >= 0)
{
da = new SqlDataAdapter("SELECT * FROM RegisterTable", con);
SqlCommandBuilder cb = new SqlCommandBuilder(da);
ds.Clear();
da.Fill(ds, "RegisterTable");
ds.Tables["RegisterTable"].PrimaryKey = new DataColumn[] { ds.Tables["RegisterTable"].Columns["Id"] };
int id = Convert.ToInt32(GridView1.SelectedDataKey.Value);
DataRow dr = ds.Tables["RegisterTable"].Rows.Find(id);
if (dr != null)
{
dr["Name"] = txtname.Text;
dr["Age"] = txtage.Text;
dr["Phone"] = txtnumber.Text;
dr["Email"] = txtemail.Text;
da.Update(ds, "RegisterTable");
BindGrid();
ClearFields();
}
}
else
{
Response.Write("<script>alert('Please select a row to update');</script>");
}
}
protected void btnDelete_Click(object sender, EventArgs e)
{
if (GridView1.SelectedIndex >= 0)
{
da = new SqlDataAdapter("SELECT * FROM RegisterTable", con);
SqlCommandBuilder cb = new SqlCommandBuilder(da);
ds.Clear();
da.Fill(ds, "RegisterTable");
ds.Tables["RegisterTable"].PrimaryKey = new DataColumn[] { ds.Tables["RegisterTable"].Columns["Id"] };
int id = Convert.ToInt32(GridView1.SelectedDataKey.Value);
DataRow dr = ds.Tables["RegisterTable"].Rows.Find(id);
if (dr != null)
{
dr.Delete();
da.Update(ds, "RegisterTable");
BindGrid();
ClearFields();
}
}
else
{
Response.Write("<script>alert('Please select a row to delete');</script>");
}
}
protected void GridView1_SelectedIndexChanged(object sender, EventArgs e)
{
txtname.Text = GridView1.SelectedRow.Cells[1].Text;
txtage.Text = GridView1.SelectedRow.Cells[2].Text;
txtnumber.Text = GridView1.SelectedRow.Cells[3].Text;
txtemail.Text = GridView1.SelectedRow.Cells[4].Text;
}
protected void btnClear_Click(object sender, EventArgs e)
{
ClearFields();
}
void ClearFields()
{
txtname.Text = txtage.Text = txtnumber.Text = txtemail.Text = "";
GridView1.SelectedIndex = -1;
}
}
}
