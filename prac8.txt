using System;
using System.Net;

namespace prac_8
{
    public partial class About : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            try
            {
                using (WebClient wc = new WebClient())
                {
                    string url = "http://127.0.0.1:5000/about";
                    string response = wc.DownloadString(url);

                    lblAbout.Text = response;
                }
            }

            catch (Exception ex)
            {
                lblAbout.Text = "Error connecting to Flask: " + ex.Message;
            }
        }
    }
}

using System;
using System.Net;
using System.Xml.Linq;

namespace prac_8
{
    public partial class User : System.Web.UI.Page
    {
        protected void btnSend_Click(object sender, EventArgs e)
        {
            string name = txtName.Text.Trim();

            if (string.IsNullOrEmpty(name))
            {
                lblResult.Text = "Please enter a name.";
                return;
            }

            try
            {
                using (WebClient wc = new WebClient())
                {
                    string url = "http://127.0.0.1:5000/user/" + name;

                    string response = wc.DownloadString(url);

                    lblResult.Text = response;
                }
            }

            catch (Exception ex)
            {
                lblResult.Text = "Error connecting to Flask: " + ex.Message;
            }
        }
    }
}
